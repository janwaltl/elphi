58 PerfEvents::get_perf_event(Buffer* dest, bool peek_only) {
59  if (!m_fd.is_opened() || m_buffer.empty() || (m_buffer.size() % c_page_size) != 0)
60  return std::nullopt;
61 
62  auto* header = type_pune<perf_event_mmap_page>(m_buffer.data());
63  // The ring buffer begins at the next page.
64  const auto buffer = m_buffer.subspan(c_page_size);
65 
66  perf_event_header event_header;
67  // Header does not fit -> no unread sample.
68  // Both values are non-decreasing.
69  if (header->data_tail + sizeof(event_header) > header->data_head)
70  return std::nullopt;
71  read_memory_barrier();
72 
73  std::span header_dest{reinterpret_cast<unsigned char*>(&event_header), sizeof(event_header)};
74  move_wrapped(buffer, header->data_tail % buffer.size(), header_dest);
75 
76  if (dest) {
77  // The event is only partially written. Can it even happen?
78  if (header->data_tail + event_header.size > header->data_head)
79  return std::nullopt;
80 
81  dest->resize(event_header.size - sizeof(perf_event_header));
82  // Some examples and manpages recommend memory barries after reading the values, not sure why.
83  read_memory_barrier();
84 
85  move_wrapped(buffer, (header->data_tail + sizeof(event_header)) % buffer.size(), *dest);
86  }
87 
88  if (!peek_only) {
89  header->data_tail += event_header.size;
90  read_memory_barrier();
91  }

94 }
